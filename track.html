function trackShipment() { let tracking = document.getElementById("trackingNumber").value.trim().toUpperCase(); if (tracking === "") { alert("Please enter a tracking number."); return; } let shipments = JSON.parse(localStorage.getItem("shipments")) || []; let shipment = shipments.find(function(s){ return s.tracking.toUpperCase() === tracking; }); if (!shipment) { alert("Tracking number not found!"); return; } document.getElementById("result").style.display = "block"; document.getElementById("trackingDisplay").textContent = shipment.tracking; document.getElementById("status").textContent = shipment.status; document.getElementById("location").textContent = shipment.location; document.getElementById("delivery").textContent = shipment.delivery; document.getElementById("route").textContent = shipment.route; document.getElementById("sender").textContent = shipment.sender; document.getElementById("receiver").textContent = shipment.receiver; document.getElementById("package").textContent = shipment.package; document.getElementById("weight").textContent = shipment.weight; document.getElementById("service").textContent = shipment.service; document.getElementById("vehicleDestination").textContent = shipment.location; let progress = shipment.progress || 0; let bar = document.getElementById("progressBar"); bar.style.width = progress + "%"; bar.innerHTML = progress + "%"; document.getElementById("history").innerHTML = shipment.history || "No shipment history available."; let timeline = document.getElementById("timeline"); timeline.innerHTML = `
Status: ${shipment.status}
📍 ${shipment.location}
Progress: ${progress}%
`; }
