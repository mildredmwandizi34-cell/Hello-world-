function trackShipment() { let tracking = document.getElementById("trackingNumber").value.trim().toUpperCase(); if (tracking === "") { alert("Please enter a tracking number."); return; } let shipments = JSON.parse(localStorage.getItem("shipments")) || []; let shipment = shipments.find(function(s) { return String(s.tracking).toUpperCase() === tracking; }); if (!shipment) { alert("Tracking number not found!"); document.getElementById("result").style.display = "none"; return; } // Show results document.getElementById("result").style.display = "block"; // Status document.getElementById("status").textContent = shipment.status || ""; document.getElementById("location").textContent = shipment.location || ""; document.getElementById("delivery").textContent = shipment.delivery || ""; document.getElementById("route").textContent = shipment.route || ""; // Details document.getElementById("trackingDisplay").textContent = shipment.tracking || ""; document.getElementById("sender").textContent = shipment.sender || ""; document.getElementById("receiver").textContent = shipment.receiver || ""; document.getElementById("package").textContent = shipment.package || ""; document.getElementById("weight").textContent = shipment.weight || ""; document.getElementById("service").textContent = shipment.service || ""; document.getElementById("vehicleDestination").textContent = shipment.location || ""; // Progress let progress = Number(shipment.progress) || 0; let bar = document.getElementById("progressBar"); bar.style.width = progress + "%"; bar.innerHTML = progress + "%"; // Route let origin = "Warehouse"; let destination = "Destination"; if (shipment.route && shipment.route.includes("→")) { let cities = shipment.route.split("→"); origin = cities[0].trim(); destination = cities[1].trim(); } document.getElementById("originCity").textContent = origin; document.getElementById("destinationCity").textContent = destination; document.getElementById("statusRoute").textContent = shipment.status; document.getElementById("routeProgress").style.width = progress + "%"; document.querySelector(".plane").style.left = progress + "%"; // Shipment History let history = document.getElementById("history"); if (shipment.history && shipment.history.trim() !== "") { history.innerHTML = shipment.history; } else { history.innerHTML = `
📦
Shipment Created
📍 ${shipment.location || ""}
🕒 ${new Date().toLocaleString()}
`; } }
